                grade = i + 1

                grade = 6
            }
        }
        return grade
    }
    
    min = lottos.filter((v,i) => win_nums.includes(v) === true).length;
    minGrade = lottoGrade(min);

    maxGrade = lottoGrade(max);
    answer = [maxGrade, minGrade]
    return answer
}
